-- sub_type querry

SELECT *
FROM gate_crash
WHERE sub_type like '%vampire%'

-- multi table sun_type querry
create view x as
SELECT *
FROM gate_crash
Where sub_type = 'angel'
union
select *
from return_ravnica
where sub_type = 'angel'

-- multi table 4 or more in inventory querry

SELECT *
FROM gate_crash
Where quantity > 3
union
select *
from return_ravnica
where quantity > 3

-- color querry

SELECT *
FROM gate_crash
Where color = 'gw'
union
select *
from return_ravnica
Where color = 'gw'

-- color and type 
SELECT *
FROM gate_crash
Where color = 'gw' AND type_ = 'Creature'
union
select *
from return_ravnica
Where color = 'gw' and type_ = 'Creature'


SELECT *
FROM gate_crash
Where color = 'bu' and rarity = 'Uncommon'
union
select *
from dragons_maze
where color = 'bu' and rarity = 'Uncommon'

-- create ravnica block
create view ravnica_block as
SELECT *
FROM gate_crash
union
select *
from dragons_maze
union
select *
from return_ravnica

select *
from ravnica_block 
where sub_type like '%vampire%'
select *
from ravnica_block 
where (color = 'b' and type_ = 'Creature') and (rarity = 'Uncommon' or rarity = 'Rare')
